             }

         for(int i=0; i< s.length() ;                     i++  ){
            String ch = s.substring(i,i+1);
            if(i%2==0){
                System.out.println(ch.toUpperCase());
            }
         }
 
 
         //Ornek 3: Verilen bir String de ilk a harfinden onceki tum characterleri console a yazdiriniz

         String s1 = "Tramway";

             if(ch=='a'){
                 break;
             }
             System.out.print(ch);
         }
